import {ApplicationModule, Component, Type} from '@angular/core';
import {MatGridListModule} from '@angular/material/grid-list';
import {PlayingFieldComponent} from "../playing-field/playing-field.component";
import {NgComponentOutlet, NgIf} from "@angular/common";
import {ChatPanelComponent} from "../chat-panel/chat-panel.component";
import {BlackCardAreaComponent} from "../black-card-area/black-card-area.component";
import {PlayerListComponent} from "../player-list/player-list.component";
import {WhiteCardsAreaComponent} from "../white-cards-area/white-cards-area.component";

/**
 Interface for a single tile intended for the grid layout made of many tiles used in game-layout.component.ts
 */
export interface Tile {
  color: string;
  cols: number;
  rows: number;
  text: string;
  /**
   This variable will hold the component associated to the specific tile, so that it can later be used for rendering inside the template
   */
  component: Type<ApplicationModule> | null;
}

/**
 * This class is responsible for creating the desired game layout by rendering a grid layout made by the MatGridListModule
 *
 * @example
 *  tiles: Tile[] = [
 *     {text: 'Tile 1', cols: 1, rows: 1, color: 'red', component: SampleAComponent},
 *     {text: 'Tile 2', cols: 1, rows: 1, color: 'green', component: SampleBComponent}
 *     ];
 *
 * @author Kevin Stosik
 *
 * @version 0.1
 */
@Component({
  selector: 'app-game-layout',
  standalone: true,
  imports: [MatGridListModule, PlayingFieldComponent, NgComponentOutlet, NgIf],
  templateUrl: './game-layout.component.html',
  styleUrl: './game-layout.component.css'
})
export class GameLayoutComponent {
  tiles: Tile[] = [
    {text: 'Black Card Area', cols: 3, rows: 4, color: 'gray', component: BlackCardAreaComponent},
    {text: 'Playing Field', cols: 12, rows: 8, color: 'lightgreen', component: PlayingFieldComponent},
    {text: 'Chat Panel', cols: 5, rows: 8, color: 'lightpink', component: ChatPanelComponent},
    {text: 'Player List', cols: 3, rows: 4, color: 'lightblue', component: PlayerListComponent},
    {text: 'White Cards Area', cols: 20, rows: 3, color: '', component: WhiteCardsAreaComponent}
  ];
}

<!-- Create a MatGridList with 20 columns and a height equal to the monitor height -->
<mat-grid-list cols="20" rowHeight="fit" gutterSize="0" style="height: 100vh;">
    @for (tile of tiles; track tile) {
    <!-- Fill the tiles with all of the relevant data -->
    <mat-grid-tile
        [colspan]="tile.cols"
        [rowspan]="tile.rows"
        [style.background]="tile.color">
        <!-- Take the ApplicationModule from tile.component and render it as a ng-container -->
        <ng-container *ngIf="tile.component; else componentNotFound">
            <ng-container *ngComponentOutlet="tile.component"></ng-container>
        </ng-container>
        <!-- Else case for when there is no component -->
        <ng-template #componentNotFound>
            <p>
                Component not found: <strong>{{tile.text}}</strong>
            </p>
        </ng-template>
    </mat-grid-tile>
    }
</mat-grid-list>
